TYPE = EXTERNAL_STAGE
STORAGE_PROVIDER = 'S3'
ENABLED = TRUE
STORAGE_AWS_ROLE_ARN = 'arn:aws:iam::396231126183:role/powerener.role'
STORAGE_ALLOWED_LOCATIONS = ('s3://powerbien.project/')
COMMENT = 'Optional Comment'

  //description Integration Object
  desc integration Powerbi_IntegrationC;

CREATE database Power;

create schema Power_Data;

create table Power_DatasetC(
Household_ID string,
Region string,
Country string,
Energy_Source string,
Monthly_Usage_kWH float,
Year int,
Household_Size int,
Income_Level string,
Urban_Rural string, Adoption_Year int, Subsidy_Received string , Cost_Savings_USD float

);


select * from Power_DatasetC;

create stage Power.Power_Data.Power_stage
url = 's3://powerbien.project/'
storage_integration = Powerbi_IntegrationC

copy into Power_DatasetC
